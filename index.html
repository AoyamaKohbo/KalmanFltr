カルマンフィルター(Kalman Filter)

動的システムを次式の離散状態方程式で与えます。

\begin{align} \mathbf{x}(n+1)&=~\mathbf{A}\mathbf{x}(n)+\mathbf{B}\mathbf{u}(n)+\mathbf{w}(n)\label{eq:RtnEqn}\\ \mathbf{y}(n)&=~\mathbf{C}\mathbf{x}(n)+\mathbf{v}(n) \end{align}

ここに、\(\mathbf{u}\)は入力ベクトル、\(\mathbf{x}\)は状態ベクトル、\(\mathbf{y}\)出力ベクトル、 \(\mathbf{A}\)は遷移行列、\(\mathbf{B}\)、\(\mathbf{C}\)は行列、\(\mathbf{v}\)出力ノイズベクトル、 \(\mathbf{w}\)は入力ノイズベクトルです。

式\eqref{eq:RtnEqn}のカルマンフィルター次式となります。

\begin{align} \hat{\mathbf{x}}(n)={\mathbf{p}}(n)+\mathbf{K}\mathbf{e}(n)\label{eq:KalmanFilt} \mathbf{e}(n)&=~\mathbf{y}(n)-\mathbf{C}\mathbf{p}(n)\\ {\mathbf{p}}(n+1)&=~\mathbf{A}\hat{\mathbf{x}}(n)+\mathbf{B}\mathbf{u}(n)\\ \end{align}

\(\mathbf{K}\)はカルマンゲイン行列であり、次式で与えられます。

\begin{align} \mathbf{K}&=~\mathbf{P}\mathbf{C}^T\mathbf{V}^{-1}\\ \mathbf{P}&=~(\mathbf{I}-\mathbf{K}\mathbf{C})(\mathbf{A}\mathbf{P}\mathbf{A}^T+\mathbf{W})\\ \mathbf{V}&=~E[\mathbf{v}(n)\mathbf{v}(n)^T]\\ \mathbf{W}&=~E[\mathbf{w}(n)\mathbf{w}(n)^T] \end{align} \begin{align} \tilde{\mathbf{x}}(n)=\mathbf{x}(n)-\hat{\mathbf{x}}(n) \end{align}

カルマンフィルターの導出

\begin{align} J=E[\tilde{\mathbf{x}}^T(n)\tilde{\mathbf{x}}(n)] \end{align} \begin{align} \mathbf{z}(n+1)&=~\mathbf{x}(n+1)-\mathbf{p}(n+1)\\ &=~\mathbf{A}\tilde{\mathbf{x}}(n)+\mathbf{w}(n) \end{align} \begin{align} \tilde{\mathbf{x}}(n)&=~\mathbf{x}(n)-\hat{\mathbf{x}}(n)\nonumber\\ &=~\mathbf{x}(n)-\mathbf{p}(n)-\mathbf{K}\{\mathbf{C}\hat{\mathbf{x}}(n)+\hat{\mathbf{v}}(n)-\hat{\mathbf{C}}\mathbf{p}(n)\}\nonumber\\ &=~(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{z}(n)-\mathbf{K}\mathbf{v}(n)\label{eq:ErrorOfEst} \end{align} \(E[\tilde{\mathbf{z}}(n)\tilde{\mathbf{v}}^T(n)]=\mathbf{0}\) \begin{align} J&=~E[{\mathbf{z}(n)^T(\mathbf{I}-\mathbf{K}\mathbf{C})^T(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{z}(n)}]+E[\mathbf{v}(n)^T\mathbf{K}^T\mathbf{K}\mathbf{v}(n)]\nonumber\\ &=~E[\mathbf{z}(n)^T\mathbf{z}(n)]-E[{\mathbf{z}(n)^T\mathbf{K}\mathbf{C}\mathbf{z}(n)}]-E[{\mathbf{z}(n)^T\mathbf{C}^T\mathbf{K}^T\mathbf{z}(n)}]+E[{\mathbf{z}(n)^T\mathbf{C}^T\mathbf{K}^T\mathbf{K}\mathbf{C}\mathbf{z}(n)}] +E[\mathbf{v}(n)^T\mathbf{K}^T\mathbf{K}\mathbf{v}(n)] \end{align} \begin{align} \frac{\partial J}{\partial\mathbf{K}}=-2E[{\mathbf{z}(n)^T\mathbf{z}(n)]\mathbf{C}}-2\mathbf{K}\mathbf{C}E[{\mathbf{z}(n)^T\mathbf{z}(n)}]\mathbf{C}+2\mathbf{K}E[\mathbf{v}(n)^T\mathbf{v}(n)]=\mathbf{0} \end{align} \begin{align} \mathbf{M}=E[\mathbf{z}(n)\mathbf{z}^T(n)] \end{align} \begin{align} \frac{\partial J}{\partial\mathbf{K}}=-2\mathbf{M}\mathbf{C}^T-2\mathbf{K}\mathbf{C}\mathbf{M}\mathbf{C}^T+2\mathbf{K}\mathbf{V}=\mathbf{0} \end{align} \begin{align} \mathbf{K}\mathbf{C}\mathbf{M}\mathbf{C}^T+\mathbf{K}\mathbf{V}=\mathbf{M}\mathbf{C}^T\label{eq:KalmanGain1} \end{align} \(E[\tilde{\mathbf{x}}(n)\tilde{\mathbf{w}}^T(n)]=\mathbf{0}\)(\eqref{eq:ErrorOfEst}) \begin{align} E[\mathbf{z}(n+1)\mathbf{z}^T(n+1)]&=~E[\{\mathbf{A}\tilde{\mathbf{x}}(n)+\mathbf{w}(n)\}\{\mathbf{A}\tilde{\mathbf{x}}(n)+\mathbf{w}(n)^T]\nonumber\\ &=~\mathbf{A}E[\tilde{\mathbf{x}}(n)\tilde{\mathbf{x}}(n)^T]\mathbf{A}^T+E[\mathbf{w}(n)\mathbf{w}(n)^T\}]\label{eq:VarMatOfZ} \end{align} \begin{align} \mathbf{P}=E[\tilde{\mathbf{x}}(n)\tilde{\mathbf{x}}^T(n)]\label{eq:CovarMatOfError} \end{align} \begin{align} \mathbf{M}=\mathbf{A}\mathbf{P}\mathbf{A}^T+\mathbf{W}\label{eq:VarMatOfZ2} \end{align} (\eqref{eq:ErrorOfEst})(\eqref{eq:CovarMatOfError}) \begin{align} \mathbf{P}=E[\{(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{z}(n)-\mathbf{K}\mathbf{v}(n)\}\{(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{z}(n)-\mathbf{K}\mathbf{v}(n)\}^T]\label{eq:CovarMatOfError2} \end{align} \(E[\mathbf{z}(n)\mathbf{v}^T(n)]=\mathbf{0}\)(\eqref{eq:CovarMatOfError2}) \begin{align} \mathbf{P}=(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{M}(\mathbf{I}-\mathbf{K}\mathbf{C})^T+\mathbf{K}\mathbf{V}\mathbf{K}^T\label{eq:MatP1} \end{align} (\eqref{eq:KalmanGain1}) \begin{align} \mathbf{K}\mathbf{V}\mathbf{K}^T=(\mathbf{I}-\mathbf{K}\mathbf{C}^T)\mathbf{M}\mathbf{C}^T\mathbf{K}^T \end{align} \begin{align} \mathbf{P}=(\mathbf{I}-\mathbf{K}\mathbf{C})\mathbf{M}\label{eq:MatP2} \end{align} \begin{align} \mathbf{P}=(\mathbf{I}-\mathbf{K}\mathbf{C})(\mathbf{A}\mathbf{P}\mathbf{A}^T+\mathbf{W}^T) \end{align} \begin{align} \mathbf{K}\mathbf{V}=\mathbf{P}\mathbf{C}^T \end{align} \begin{align} \mathbf{K}=\mathbf{P}\mathbf{C}^T\mathbf{V}^{-1} \end{align}
